67 int get_cmd_line(DWORD dwId, PWSTR &buf)
68 {
69  try
70  {
71  HANDLE hProcess = OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, FALSE, dwId);
72  DWORD err = 0;
73  if (hProcess == NULL)
74  {
75  printf("OpenProcess %u failed\n", dwId);
76  return GetLastError();
77  }
78 
79  // determine if 64 or 32-bit processor
80  SYSTEM_INFO si;
81  GetNativeSystemInfo(&si);
82 
83  // determine if this process is running on WOW64
84  BOOL wow;
85  IsWow64Process(GetCurrentProcess(), &wow);
86 
87  // use WinDbg "dt ntdll!_PEB" command and search for ProcessParameters offset to find the truth out
88  DWORD ProcessParametersOffset = si.wProcessorArchitecture == PROCESSOR_ARCHITECTURE_AMD64 ? 0x20 : 0x10;
89  DWORD CommandLineOffset = si.wProcessorArchitecture == PROCESSOR_ARCHITECTURE_AMD64 ? 0x70 : 0x40;
90 
91  // read basic info to get ProcessParameters address, we only need the beginning of PEB
92  DWORD pebSize = ProcessParametersOffset + 8;
93  PBYTE peb = (PBYTE)malloc(pebSize);
94  ZeroMemory(peb, pebSize);
95 
96  // read basic info to get CommandLine address, we only need the beginning of ProcessParameters
97  DWORD ppSize = CommandLineOffset + 16;
98  PBYTE pp = (PBYTE)malloc(ppSize);
99  ZeroMemory(peb, pebSize);
100 
101  PWSTR cmdLine;
102 
103 
104  if (wow)
105  {
106  // we're running as a 32-bit process in a 64-bit OS
107  PROCESS_BASIC_INFORMATION_WOW64 pbi;
108  ZeroMemory(&pbi, sizeof(pbi));
109 
110  // get process information from 64-bit world
111  _NtQueryInformationProcess query = (_NtQueryInformationProcess)GetProcAddress(GetModuleHandleA("ntdll.dll"), "NtWow64QueryInformationProcess64");
112  err = query(hProcess, 0, &pbi, sizeof(pbi), NULL);
113  if (err != 0)
114  {
115  printf("NtWow64QueryInformationProcess64 failed\n");
116  CloseHandle(hProcess);
117  return GetLastError();
118  }
119 
120  // read PEB from 64-bit address space
121  _NtWow64ReadVirtualMemory64 read = (_NtWow64ReadVirtualMemory64)GetProcAddress(GetModuleHandleA("ntdll.dll"), "NtWow64ReadVirtualMemory64");
122  err = read(hProcess, pbi.PebBaseAddress, peb, pebSize, NULL);
123  if (err != 0)
124  {
125  printf("NtWow64ReadVirtualMemory64 PEB failed\n");
126  CloseHandle(hProcess);
127  return GetLastError();
128  }
129 
130  // read ProcessParameters from 64-bit address space
131  PBYTE* parameters = (PBYTE*)*(LPVOID*)(peb + ProcessParametersOffset); // address in remote process adress space
132  err = read(hProcess, parameters, pp, ppSize, NULL);
133  if (err != 0)
134  {
135  printf("NtWow64ReadVirtualMemory64 Parameters failed\n");
136  CloseHandle(hProcess);
137  return GetLastError();
138  }
139 
140  // read CommandLine
141  UNICODE_STRING_WOW64* pCommandLine = (UNICODE_STRING_WOW64*)(pp + CommandLineOffset);
142  cmdLine = (PWSTR)malloc(pCommandLine->MaximumLength);
143  err = read(hProcess, pCommandLine->Buffer, cmdLine, pCommandLine->MaximumLength, NULL);
144  if (err != 0)
145  {
146  printf("NtWow64ReadVirtualMemory64 Parameters failed\n");
147  CloseHandle(hProcess);
148  return GetLastError();
149  }
150  }
151  else
152  {
153  // we're running as a 32-bit process in a 32-bit OS, or as a 64-bit process in a 64-bit OS
154  PROCESS_BASIC_INFORMATION pbi;
155  ZeroMemory(&pbi, sizeof(pbi));
156 
157  // get process information
158  _NtQueryInformationProcess query = (_NtQueryInformationProcess)GetProcAddress(GetModuleHandleA("ntdll.dll"), "NtQueryInformationProcess");
159  err = query(hProcess, 0, &pbi, sizeof(pbi), NULL);
160  if (!err)
161  {
162  printf("NtQueryInformationProcess failed\n");
163  CloseHandle(hProcess);
164  return GetLastError();
165  }
166 
167  // read PEB
168  if (!ReadProcessMemory(hProcess, pbi.PebBaseAddress, peb, pebSize, NULL))
169  {
170  printf("ReadProcessMemory PEB failed\n");
171  CloseHandle(hProcess);
172  return GetLastError();
173  }
174 
175  // read ProcessParameters
176  PBYTE* parameters = (PBYTE*)*(LPVOID*)(peb + ProcessParametersOffset); // address in remote process adress space
177  if (!ReadProcessMemory(hProcess, parameters, pp, ppSize, NULL))
178  {
179  printf("ReadProcessMemory Parameters failed\n");
180  CloseHandle(hProcess);
181  return GetLastError();
182  }
183 
184  // read CommandLine
185  UNICODE_STRING* pCommandLine = (UNICODE_STRING*)(pp + CommandLineOffset);
186  cmdLine = (PWSTR)malloc(pCommandLine->MaximumLength);
187  if (!ReadProcessMemory(hProcess, pCommandLine->Buffer, cmdLine, pCommandLine->MaximumLength, NULL))
188  {
189  printf("ReadProcessMemory Parameters failed\n");
190  CloseHandle(hProcess);
191  return GetLastError();
192  }
193  }
194 
195  buf = cmdLine;
196  }
197  catch (exception())
198  {
199  cout << "command line function fails" << endl;
200  }
